///객체사용///
class Blink {
    private :
        long count = 0;
        long restCount = 0;
        long blinkRhythm[2] = {0b1010101010101001010100101010, 0b101010010010};
        long blinkTemp= 0;
        int pinNumber=0;
        int pattern=0;
    public:
        Blink(int _pinNumber, int _pattern){
            pinNumber = _pinNumber;
            pattern = _pattern;
        }
          
    
        void blinkRun() {
        if (count++ < 100000) return blinkRun();
        count = 0;

        digitalWrite(pinNumber, blinkTemp & 1);
        blinkTemp = blinkTemp >> 1;
        if (blinkTemp == 0) {
            restCount++;
            if (restCount == 4) {
                blinkTemp = blinkRhythm[pattern];
                restCount = 0;
            }
        }
        pinMode(pinNumber, OUTPUT);
    }
};

Blink b1(13,0);
Blink b2(12,0);
Blink b3(8,1);

void loop() {
   
    b1.blinkRun();
    b2.blinkRun();
    b3.blinkRun();
}

void setup() {
pinMode(10, OUTPUT);
}
