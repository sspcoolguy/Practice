///////////////////////////
void pin13(void) {
  static const int PIN = 13;
  pinMode(PIN, OUTPUT);

  static long counter = 0;

  if (counter++ < 50000) return;
  counter = 0;
  static bool isTrue = false;
  isTrue = !isTrue;

  if (isTrue) {
    static bool ledState = HIGH;
    digitalWrite(PIN, ledState);
    ledState = !ledState;
  }
}


void setup() {}

void loop()
{
  pin13();
}
