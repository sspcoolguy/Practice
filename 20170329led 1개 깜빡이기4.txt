void blinkLED(const int PIN) {

  static bool ledState = HIGH;

  digitalWrite(PIN, ledState);
  ledState = !ledState;
}

bool count(void) {

  static long counter = 0;
  static bool isTrue = false;

  if(counter++ < 50000) return false;
  
  counter = 0;
  isTrue = !isTrue;

  return isTrue;
}

void pin13(void){
  pinMode(13, OUTPUT);
  blinkLED(13);
}

void setup(){}


void loop()
{
  if(count()) pin13();
}
