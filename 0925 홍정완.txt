int t = 20000;
int pin13 = 13;

void setup(){
	pinMode(pin13, OUTPUT);
}

void loop() {
	Flash337(t);	  
}

void Flash337(const long DELAY_TIME){
	static int count = 0;
	static const int BAKSU = 0b101010100010101000101010;
	static int baksuTemp = 0;

	if(count++ < DELAY_TIME) return;
	count = 0;
	
	digitalWrite(13,baksuTemp & 0b1);
	baksuTemp = baksuTemp >> 1;
	if(baksuTemp == 0) {
		baksuTemp = BAKSU;
	}
}
